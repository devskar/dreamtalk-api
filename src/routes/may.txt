import express from 'express';

const router = express.Router();

router.get('/', async (req, res, next) => {

    const allTodos = await pool.query('SELECT * FROM dreams');
    res.json(allTodos.rows);
  } catch (err) {

  }

// CREATE A NEW DREAM
router.post('/', async (req, res, next) => {
  try {
    const { content } = req.body;

    const newDream = await pool.query(
      'INSERT INTO dreams (content) VALUES($1) RETURNING *',
      [content]
    );
    res.status(200).json(newDream.rows[0]);
  } catch (err) {
    console.error(err.message);
  }
});

// GET A SPECIFIC DREAM BY ID
router.get('/:id', async (req, res, next) => {
  try {
    const dream = await pool.query('SELECT * FROM dreams WHERE dream_id = $1', [
      req.params.id,
    ]);
    res.json(dream.rows[0]);
  } catch (err) {
    console.error(err.message);
  }
});

// UPDATE A SPECIFIC DREAM BY ID
router.put('/:id', async (req, res, next) => {
  try {
    const { id } = req.params;

    var values = [];
    const temp = [];

    for (let key in req.body) {
      // makes sure only if key is 'content' or 'title'
      if (['title', 'content'].includes(key)) {
        temp.push(key + ` = \$${values.length + 1}`);

        values.push(req.body[key]);
      }
    }

    // pushing the id at the end so we can access it by $i which is the last element of the array
    values.push(String(id));

    console.log(values, values.length);

    if (values.length !== 0) {
      let sql_sequence = temp.toString();

      const updatedDream = await pool.query(
        `UPDATE dreams SET ${sql_sequence} WHERE dream_id = \$${values.length} RETURNING *`,
        values
      );

      res.status(200).json(updatedDream.rows[0]);
    }
  } catch (err) {
    console.error(err.message);
  }
});

// DELETE A DREAM BY ID

export default router;
